from pyspark.sql import SparkSession
from pyspark.sql.functions import col, avg, max, min, count

# -------------------------------------------------------
# 1. Create Spark Session
# -------------------------------------------------------
spark = SparkSession.builder \
    .appName("IrisDataFrameOperations") \
    .master("local[*]") \
    .getOrCreate()

# -------------------------------------------------------
# 2. Iris Dataset (embedded - no external file needed)
# -------------------------------------------------------
iris_data = [
    (5.1,3.5,1.4,0.2,"setosa"),
    (4.9,3.0,1.4,0.2,"setosa"),
    (4.7,3.2,1.3,0.2,"setosa"),
    (4.6,3.1,1.5,0.2,"setosa"),
    (5.0,3.6,1.4,0.2,"setosa"),
    (5.4,3.9,1.7,0.4,"setosa"),
    (7.0,3.2,4.7,1.4,"versicolor"),
    (6.4,3.2,4.5,1.5,"versicolor"),
    (6.9,3.1,4.9,1.5,"versicolor"),
    (5.5,2.3,4.0,1.3,"versicolor"),
    (6.5,2.8,4.6,1.5,"versicolor"),
    (6.3,3.3,6.0,2.5,"virginica"),
    (5.8,2.7,5.1,1.9,"virginica"),
    (7.1,3.0,5.9,2.1,"virginica"),
    (6.3,2.9,5.6,1.8,"virginica"),
    (6.5,3.0,5.8,2.2,"virginica")
]

columns = ["sepal_length","sepal_width","petal_length","petal_width","species"]

# Create DataFrame
df = spark.createDataFrame(iris_data, schema=columns)

print("\n================ IRIS DATAFRAME =================")
df.show()

# -------------------------------------------------------
# 3. Select Operation
# -------------------------------------------------------
print("\n1. SELECT SPECIFIC COLUMNS")
df.select("sepal_length", "species").show()

# -------------------------------------------------------
# 4. Filter Operation
# -------------------------------------------------------
print("\n2. FILTER: Petal Length > 5")
df.filter(col("petal_length") > 5).show()

# -------------------------------------------------------
# 5. Add New Column (withColumn)
# -------------------------------------------------------
print("\n3. ADD COLUMN: Sepal Area")
df_with_area = df.withColumn(
    "sepal_area",
    col("sepal_length") * col("sepal_width")
)

df_with_area.show()

# -------------------------------------------------------
# 6. GroupBy + Aggregations
# -------------------------------------------------------
print("\n4. GROUP BY SPECIES - AVG VALUES")
df.groupBy("species").agg(
    avg("sepal_length").alias("avg_sepal_length"),
    avg("petal_length").alias("avg_petal_length"),
    count("*").alias("count")
).show()

# -------------------------------------------------------
# 7. MAX & MIN per Species
# -------------------------------------------------------
print("\n5. MAX PETAL LENGTH PER SPECIES")
df.groupBy("species").agg(
    max("petal_length").alias("max_petal_length")
).show()

print("\n6. MIN SEPAL WIDTH PER SPECIES")
df.groupBy("species").agg(
    min("sepal_width").alias("min_sepal_width")
).show()

# -------------------------------------------------------
# 8. Sorting / OrderBy
# -------------------------------------------------------
print("\n7. ORDER BY PETAL LENGTH DESC")
df.orderBy(col("petal_length").desc()).show()

# -------------------------------------------------------
# 9. Temporary SQL View
# -------------------------------------------------------
df.createOrReplaceTempView("iris")

print("\n8. SQL QUERY: AVG SEPAL LENGTH")
spark.sql("""
    SELECT species, AVG(sepal_length) as avg_sepal
    FROM iris
    GROUP BY species
""").show()

# -------------------------------------------------------
# Stop Spark
# -------------------------------------------------------
spark.stop()
